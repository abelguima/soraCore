# copy macro, will create empty dest file if src doesn't exist
macro( filecopy SRC_FILE DST_FILE )
  if( EXISTS ${SRC_FILE} )
    exec_program("${CMAKE_COMMAND}" ARGS 
                  -E copy_if_different        
                  ${SRC_FILE}  
                  ${DST_FILE} 
    )
  else( EXISTS ${SRC_FILE} )
    file( WRITE ${DST_FILE} "" )
  endif( EXISTS ${SRC_FILE} )
endmacro( filecopy )
